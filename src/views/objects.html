Categories of objects:

* Ordinary objects: All the default internal behaviors for objects in javascript.

* Exotic objects: Internal behavior, differs from the default in some ways.

* Standard objects: Defined by ECMAScript 6, Array, Date and some others may be ordinary or exotic.

* Built-in objects: They are present in javascript execution environment when javascript starts executing, all standard objects are built-in objects.

Property initializer shorthand:

Initializer shorthand eliminates the duplication of properties when a object property name is the same as the local variable name.

Without initializer shorthand

function createPerson(name, age) { 
    return { 
        name: name, 
        age: age 
    }; 
}

With initializer shorthand

function createPerson(name, age) { 
    return { 
        name, 
        age 
    }; 
}
Concise method syntax:

The shorthand syntax is also called concise method syntax, the property is assigned to an anonymous function.

Concise method may use "super" (we will check this later).

You also delete the "function" word

Without concise method syntax:

var person = { 
    name: "Nicholas", 
    sayName: function() { 
        console.log(this.name); 
    } 
};

With concise method syntax:

var person = { 
    name: "Nicholas", 
    sayName() { 
        console.log(this.name); 
    } 
};
You can work with objects in many ways, here are some examples:

* Example 1:

var person = {}, 
    lastName = "last name"; 
 
person["first name"] = "Nicholas"; 
person[lastName] = "Zakas"; 
 
console.log(person["first name"]);      // "Nicholas" 
console.log(person[lastName]);          // "Zakas"

* Example 2:

var person = { 
    "first name": "Nicholas" 
}; 
 
console.log(person["first name"]);      // "Nicholas"

* Example 3:

var lastName = "last name"; 
var person = { 
    "first name": "Nicholas", 
    [lastName]: "Zakas" 
}; 
    
console.log(person["first name"]);      // "Nicholas" 
console.log(person[lastName]);          // "Zakas"

* Example 4:

var suffix = " name";  
var person = { 
    ["first" + suffix]: "Nicholas", 
    ["last" + suffix]: "Zakas" 
}; 
    
console.log(person["first name"]);      // "Nicholas" 
console.log(person["last name"]);       // "Zakas"
Object.is():

A better comparisson method than the used by ordinary javascript ( == and === ), some of the examples are:

In traditional javascript (+0 and -0) are equal, (NaN and NaN) are different, (5 and "5") are equal, and those 3 examples are wrong.

Object.is() fixes those problems and we have this as a result:

console.log(+0 == -0);              // true 
console.log(+0 === -0);             // true 
console.log(Object.is(+0, -0));     // false 
    
console.log(NaN == NaN);            // false 
console.log(NaN === NaN);           // false 
console.log(Object.is(NaN, NaN));   // true 
    
console.log(5 == 5);                // true 
console.log(5 == "5");              // true 
console.log(5 === 5);               // true 
console.log(5 === "5");             // false 
console.log(Object.is(5, 5));       // true 
console.log(Object.is(5, "5"));     // false

Object.assign has the same behavior as the Mixins, they can accept any number of suppliers and the receiver receives the properties in the order in which the suppliers are specified, example: 

var receiver = {};
Object.assign(receiver,
    {
        type: "js",
        name: "file.js"
    },
    {
        type: "css"
    }
);

console.log(receiver.type);     // "css"
console.log(receiver.name);     // "file.js"

This duplicated property "type" has the value of the last asignation.

GetOwnPropertyNames() method:

This method gives me the order of the keys in an object, first the numbers in ascending, then the string keys and at the end the symbols, example: 

var obj = { 
    a: 1, 
    0: 1, 
    c: 1, 
    2: 1, 
    b: 1, 
    1: 1 
}; 
 
obj.d = 1; 
 
console.log(Object.getOwnPropertyNames(obj).join(""));     // "012acbd"
Prototypes:

As we said before, prototypes allows to inherit an object into another variable, so you can have the same object structure with methods in another var, and we have two different methods, object.getPrototypeOf() and object.setPrototypeOf(), example: 

let person = { 
    getGreeting() { 
        return "Hello"; 
    } 
}; 
 
let dog = { 
    getGreeting() { 
        return "Woof"; 
    } 
}; 
 
 
let friend = { 
    getGreeting() { 
        return Object.getPrototypeOf(this).getGreeting.call(this) + ", hi!"; 
    } 
}; 
 
// set prototype to person 
Object.setPrototypeOf(friend, person); 
console.log(friend.getGreeting());                      // "Hello, hi!" 
console.log(Object.getPrototypeOf(friend) === person);  // true 
 
// set prototype to dog 
Object.setPrototypeOf(friend, dog); 
console.log(friend.getGreeting());                      // "Woof, hi!" 
console.log(Object.getPrototypeOf(friend) === dog);     // true
Super value:

The super value replaces the object.getPrototypeOf(this), we will be using this in more blocks, but here is an example: 

let friend = { 
    getGreeting() { 
        // in the previous example, this is the same as: 
        // Object.getPrototypeOf(this).getGreeting.call(this) 
        return super.getGreeting() + ", hi!"; 
    } 
};
